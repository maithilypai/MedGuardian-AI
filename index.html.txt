    <script>
        function checkHealth() {
            let heartRate = Math.floor(Math.random() * 50) + 60;
            let oxygen = Math.floor(Math.random() * 10) + 90;

            let status = "Normal ";

            if (heartRate > 100 || oxygen < 92) {
                status = "High Risk ";
            }

            document.getElementById("result").innerHTML =
                "Heart Rate: " + heartRate + " bpm <br>" +
                "Oxygen: " + oxygen + "% <br>" +
                "Status: " + status;
        }
    </script>
